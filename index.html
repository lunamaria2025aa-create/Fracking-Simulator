def mostrar_interfaz(trimestre, riesgo, social, produccion_total): print("\n" + "="*40) print(f" SIMULADOR FRACKING - T{trimestre}") print("="*40) print(f" [ESTADO] Riesgo: {riesgo:.1f}% | Social: {social:.1f}%") print(f" [PROD] Total acumulado: {produccion_total:.1f}") print("-"*40) def simulador_limpio(): riesgo = 0.0 social = 100.0 produccion_total = 0.0 for t in range(1, 11): mostrar_interfaz(t, riesgo, social, produccion_total) # Entradas del usuario try: print("Asignar presupuesto (Suma máx 100):") tec = float(input(" -> Tecnología: ")) mon = float(input(" -> Monitoreo: ")) com = float(input(" -> Comunidad: ")) prod = float(input(" -> Perforar: ")) except ValueError: print("Error: Ingresa solo números.") continue # Lógica de simulación riesgo = max(0, riesgo + (prod * 0.7) - (tec * 0.4) - (mon * 0.4)) social = max(0, social + (com * 0.5) - (riesgo * 0.1)) produccion_total += (prod * 1.2) if riesgo > 70: print("\n!!! ALERTA: CONDICIONES CRÍTICAS !!!") break print("\n" + "="*40) print("FIN DE LA SIMULACIÓN") print(f"Producción Final: {produccion_total:.1f}") print("="*40) simulador_limpio()
